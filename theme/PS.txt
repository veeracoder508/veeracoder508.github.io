        <button onclick="convertNumber()">Convert</button>

    <script>
        function convertNumber() {
            const decimalInput = document.getElementById("decimalInput");
            const binaryOutput = document.getElementById("binaryOutput");
            const hexOutput = document.getElementById("hexOutput");
            const octalOutput = document.getElementById("octalOutput");

            const decimalValue = parseInt(decimalInput.value);

            if (isNaN(decimalValue)) {
                binaryOutput.textContent = "Invalid input. Please enter a number.";
                hexOutput.textContent = "";
                octalOutput.textContent = "";
                return;
            }

            binaryOutput.textContent = decimalValue.toString(2);
            hexOutput.textContent = decimalValue.toString(16).toUpperCase(); // Uppercase for hex
            octalOutput.textContent = decimalValue.toString(8);
        }
    </script>
